# plot windowed signal and fourier transform

def windowed_ft(t, x, Fs, w_pos_sec, w_len, w_type, upper_y=1.0):
    
    N = len(x)
    w_pos = int(Fs * w_pos_sec)
    w = np.zeros(N)
    w[w_pos:w_pos + w_len] = scipy.signal.get_window(w_type, w_len)
    x = x * w
    
    plt.figure(figsize=(15, 2))

    plt.subplot(1, 2, 1)
    plt.plot(t, x, c='k')
    plt.plot(t, w, c='r')
    plt.xlim([min(t), max(t)])
    plt.ylim([-1.1, 1.1])
    plt.xlabel('Time (seconds)')

    plt.subplot(1, 2, 2)
    X = np.abs(np.fft.fft(x)) / N * 2
    freq = np.fft.fftfreq(N, d=1/Fs)
    X = X[:N//2]
    freq = freq[:N//2]
    plt.plot(freq, X, c='k')
    plt.xlim([0, 50])
    plt.ylim([0, upper_y])
    plt.xlabel('Frequency (Hz)')
    
w_len = 512
w_pos = 1280
windowed_ft(t, x, Fs, 1.0, w_len, 'boxcar', upper_y=0.15)
windowed_ft(t, x, Fs, 1.0, w_len, 'triang', upper_y=0.15)
windowed_ft(t, x, Fs, 1.0, w_len, 'hann', upper_y=0.15)

all_windows = 'boxcar triang blackman hamming hann bartlett flattop parzen bohman blackmanharris nuttall barthann'.split()
interact(windowed_ft,
         w_pos_sec=FloatSlider(min=0, max=duration-(w_len/Fs), step=0.1, continuous_update=False, value=1.0),
         w_type=Dropdown(options=all_windows),
         t=fixed(t), x=fixed(x), Fs=fixed(Fs), w_len=fixed(w_len), upper_y=fixed(0.15))
